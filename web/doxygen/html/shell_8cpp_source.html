125 const bool Shell::OnStartExecution(void)
126 {
127  GetTraces().Trace(TRACE_SHELL) << "Shell for CLI '" << GetCli().GetKeyword() << "' starts running." << endl;
128 
129  // Check the CLI is the first menu.
130  if (m_qMenus.GetHead() == & m_cliCli)
131  {
132  // Enter the shell.
133  PromptWelcomeMessage();
134  PromptMenu();
135 
136  return true;
137  }

142 const bool Shell::OnStopExecution(void)
143 {
144  bool b_Res = true;
145 
146  // Remove all menus of the menu stack to let the shell quit.
147  while (! m_qMenus.IsEmpty())
148  {
149  ExitMenu(false);
150  }
151 
152  // Leave the shell.
153  PromptByeMessage();
154 
155  // Restore the CLI reference.
156  if (! m_qMenus.AddTail(& m_cliCli))
157  {
158  GetTraces().Trace(INTERNAL_ERROR) << "Could not restore a valid status on shell exit." << endl;
159  b_Res = false;
160  }
161 
162  GetTraces().Trace(TRACE_SHELL) << "Shell for CLI '" << GetCli().GetKeyword() << "' end of run." << endl;
163  return b_Res;
164 }

211 void Shell::PromptMenu(void) const
212 {
213  if (IsRunning() && (! m_qMenus.IsEmpty()))
214  {
215  ExecutionContext* const pcli_CurrentContext = const_cast<Shell*>(this)->GetContextManager().GetCurrentContext();
216  if ((pcli_CurrentContext != NULL) && (pcli_CurrentContext != this))
217  {
218  // Non-blocking device and context switching management.
219  class WaitForPrompt : public ExecutionResult {
220  private:
221  const Shell& m_cliShell;
222  public:
223  explicit WaitForPrompt(const Shell& CLI_Shell) : m_cliShell(CLI_Shell) {}
224  public: virtual void OnResult(const ExecutionContext& CLI_Context) { m_cliShell.PromptMenu(); delete this; }
225  };
226  if (WaitForPrompt* const pcli_WaitForPrompt = new WaitForPrompt(*this))
227  {
228  pcli_WaitForPrompt->WatchResult(*pcli_CurrentContext);
229  }
230  }
231  else
232  {
233  // Show the prompt.
234  const tk::String str_NoDefaultPrompt = m_cliNoDefaultPrompt.GetString(GetLang());
235  if (! str_NoDefaultPrompt.IsEmpty())
236  {
237  GetStream(PROMPT_STREAM) << str_NoDefaultPrompt;
238  }
239  else
240  {
241  if (const Menu* const pcli_Menu = m_qMenus.GetTail())
242  {
243  const tk::String cli_Prompt = pcli_Menu->OnPrompt();
244  if (! cli_Prompt.IsEmpty())
245  {
246  GetStream(PROMPT_STREAM) << cli_Prompt << ">";
247  }
248  else
249  {
250  GetStream(PROMPT_STREAM) << pcli_Menu->GetKeyword() << ">";
251  }
252  }
253  }
254 
255  // Eventually echo the current line.
256  m_cliCmdLine.PrintCmdLine(GetStream(ECHO_STREAM));
257  }
258  }
259 }

310 void Shell::DisplayHelp(void)
311 {
312  if (! m_qMenus.IsEmpty())
313  {
314  // First display comment line patterns if any.
315  if (! GetCli().GetCommentLinePatterns().IsEmpty())
316  {
317  for (unsigned int i = 0; i < HELP_MARGIN; i++)
318  {
319  GetStream(OUTPUT_STREAM) << " ";
320  }
321  GetStream(OUTPUT_STREAM) << "Start with ";
322  unsigned int ui_PatternsDisplayed = 0;
323  for ( tk::Queue<tk::String>::Iterator it = GetCli().GetCommentLinePatterns().GetIterator();
324  GetCli().GetCommentLinePatterns().IsValid(it);
325  GetCli().GetCommentLinePatterns().MoveNext(it))
326  {
327  ui_PatternsDisplayed ++;
328  if (ui_PatternsDisplayed == GetCli().GetCommentLinePatterns().GetCount())
329  {
330  GetStream(OUTPUT_STREAM) << "or ";
331  }
332  GetStream(OUTPUT_STREAM) << "'" << GetCli().GetCommentLinePatterns().GetAt(it) << "'";
333  if (ui_PatternsDisplayed < GetCli().GetCommentLinePatterns().GetCount())
334  {
335  if (ui_PatternsDisplayed + 1 < GetCli().GetCommentLinePatterns().GetCount())
336  {
337  GetStream(OUTPUT_STREAM) << ",";
338  }
339  GetStream(OUTPUT_STREAM) << " ";
340  }
341  else
342  {
343  GetStream(OUTPUT_STREAM) << " for a comment line" << endl;
344  }
345  }
346  }
347 
348  const tk::String tk_Left = m_cliCmdLine.GetLeft(), tk_Right = m_cliCmdLine.GetRight();
349  m_cliCmdLine.Reset();
350  OnHelp(
351  false, // Execution within the context of a command line processing. OnExecute() does already prompt the menu.
352  false // Just help, no completion
353  );
354  m_cliCmdLine.Set(tk_Left, tk_Right);
355  }
356 }

415 void Shell::OnKey(const KEY E_KeyCode)
416 {
417  if ((! m_qMenus.IsEmpty()) // Menu stack control.
418  // If there are menus in the menu stack,
419  // it means we are still waiting for command lines.
420  && (E_KeyCode != NULL_KEY)) // End of input.
421  {
422  // In general, history navigation is lost.
423  // Except for navigation keys, in history, and within the line.
424  switch (E_KeyCode)
425  {
426  case KEY_UP: case KEY_DOWN: case PAGE_UP: case PAGE_DOWN:
427  case KEY_BEGIN: case KEY_END: case KEY_LEFT: case KEY_RIGHT:
428  break;
429  default:
430  m_cliHistory.EnableNavigationMemory(false);
431  break;
432  }
433 
434  switch (E_KeyCode)
435  {
436  case KEY_UP: OnHistory(1); break;
437  case KEY_DOWN: OnHistory(-1); break;
438  case PAGE_UP: OnHistory(HISTORY_PAGE); break;
439  case PAGE_DOWN: OnHistory(- (signed int) HISTORY_PAGE); break;
440  case KEY_BEGIN: OnKeyBegin(); break;
441  case KEY_END: OnKeyEnd(); break;
442  case KEY_LEFT: OnKeyLeft(); break;
443  case KEY_RIGHT: OnKeyRight(); break;
444  case BACKSPACE: OnBackspace(); break;
445  case DELETE: OnSuppr(); break;
446  case ENTER: OnExecute(); break;
447  case BREAK:
448  case ESCAPE: OnEscape(); break;
449  case LOGOUT: OnExit(true); break;
450  case TAB: OnHelp(true, true); break;
451  case QUESTION:
452  if (m_cliCmdLine.GetLine().GetChar(m_cliCmdLine.GetLine().GetLength() - 1) == '\\')
453  {
454  OnPrintableChar(E_KeyCode);
455  }
456  else
457  {
458  OnHelp(true, false);
459  }
460  break;
461  case CLS: CleanScreen(true); break;
462 
463  case KEY_a: case KEY_aacute: case KEY_agrave: case KEY_auml: case KEY_acirc:
464  case KEY_b: case KEY_c: case KEY_ccedil: case KEY_d:
465  case KEY_e: case KEY_eacute: case KEY_egrave: case KEY_euml: case KEY_ecirc:

471  case KEY_u: case KEY_uacute: case KEY_ugrave: case KEY_uuml: case KEY_ucirc:
472  case KEY_v: case KEY_w: case KEY_x: case KEY_y: case KEY_z:

477  case KEY_S: case KEY_T: case KEY_U: case KEY_V: case KEY_W: case KEY_X:
478  case KEY_Y: case KEY_Z:
479 
480  case KEY_0: case KEY_1: case KEY_2: case KEY_3: case KEY_4: case KEY_5:

626 void Shell::OnHelp(const bool B_PromptMenu, const bool B_Completion)
627 {
628  if (! m_qMenus.IsEmpty())
629  {
630  // Parse the command line.
631  CommandLine cli_CommandLine;
632  if (cli_CommandLine.Parse(*m_qMenus.GetTail(), m_cliCmdLine.GetLeft(), false))
633  {
634  // Research matching elements.
635  Element::List cli_Exact(MAX_WORDS_PER_NODE), cli_Elements(MAX_WORDS_PER_NODE);
636  if (! cli_CommandLine.GetLastElement().FindElements(
637  cli_Exact, cli_Elements, cli_CommandLine.GetLastWord()))
638  {
639  return;
640  }
641 
642  // Initialize completion so far.
643  tk::String str_CompletionSoFar(MAX_WORD_LENGTH);
644  // Determine whether help should be printed, and completion activated.
645  bool b_PrintHelpList = false;
646 
647  if (! B_Completion)
648  {
649  // Just print help.
650  b_PrintHelpList = true;
651  }
652  else
653  {
654  // Print help and do completion depending on the possible nodes.
655  if (cli_Elements.IsEmpty())
656  {
657  // No matching element.
658  Beep();
659  }
660  else if ((cli_Elements.GetCount() == 1)
661  && (dynamic_cast<const Endl*>(cli_Elements.GetHead()) == NULL)
662  && (dynamic_cast<const Param*>(cli_Elements.GetHead()) == NULL))
663  {
664  // Straight forward completion.
665  m_cliCmdLine.Delete(GetStream(ECHO_STREAM), - cli_CommandLine.GetNumBackspacesForCompletion());
666  m_cliCmdLine.Put(GetStream(ECHO_STREAM), cli_Elements.GetHead()->GetKeyword());
667  m_cliCmdLine.Put(GetStream(ECHO_STREAM), SPACE);
668  }
669  else
670  {
671  // Look for completion so far.
672  bool b_InitCompletionSoFar = true;
673  for ( Element::List::Iterator it = cli_Elements.GetIterator();
674  cli_Elements.IsValid(it);
675  cli_Elements.MoveNext(it))
676  {
677  if ((dynamic_cast<const Param*>(cli_Elements.GetAt(it)) != NULL)
678  || (dynamic_cast<const Endl*>(cli_Elements.GetAt(it)) != NULL))
679  {
680  // No completion for parameters and end of lines.
681  str_CompletionSoFar.Set("");
682  break;
683  }
684  else if (const Element* const pcli_Element = dynamic_cast<const Element*>(cli_Elements.GetAt(it)))
685  {
686  if (b_InitCompletionSoFar)
687  {
688  // First initialization of str_CompletionSoFar.
689  if (! str_CompletionSoFar.Set(pcli_Element->GetKeyword()))
690  GetTraces().Trace(INTERNAL_ERROR) << "Shell::OnHelp(): Could not store completion so far" << endl;
691  b_InitCompletionSoFar = false;
692  }
693  else
694  {
695  // Remove last character of str_CompletionSoFar till it matches the current keyword.
696  while ( (! str_CompletionSoFar.IsEmpty())
697  && (pcli_Element->GetKeyword().SubString(0, str_CompletionSoFar.GetLength()) != str_CompletionSoFar))
698  {
699  if (! str_CompletionSoFar.Set(str_CompletionSoFar.SubString(0, str_CompletionSoFar.GetLength() - 1)))
700  GetTraces().Trace(INTERNAL_ERROR) << "Shell::OnHelp(): Could not store completion so far" << endl;
701  }
702  }
703  }
704  }
705 
706  // When no more completion so far available, print help.
707  if ((cli_CommandLine.GetLastWord() == NULL)
708  || (str_CompletionSoFar == cli_CommandLine.GetLastWord()))
709  {
710  b_PrintHelpList = true;
711  }
712  }
713  }
714 
715  // Help display.
716  if (b_PrintHelpList)
717  {
718  // First of all, sort help elements on keywords.
719  class _ { public:
720  static const int cmp(const Element* const& PCLI_1, const Element* const& PCLI_2)
721  {
722  if (PCLI_1->GetKeyword() < PCLI_2->GetKeyword()) { return 1; }
723  if (PCLI_1->GetKeyword() > PCLI_2->GetKeyword()) { return -1; }
724  return 0;
725  }
726  };
727  cli_Elements.Sort(_::cmp);
728 
729  if (B_PromptMenu)
730  {
731  m_cliCmdLine.NextLine(GetStream(ECHO_STREAM));
732  }
733 
734  // For each matching keyword...
735  for ( Element::List::Iterator it = cli_Elements.GetIterator();
736  cli_Elements.IsValid(it);
737  cli_Elements.MoveNext(it))
738  {
739  // Print the corresponding help.
740  PrintHelp(*cli_Elements.GetAt(it));
741  }
742 
743  if (B_PromptMenu)
744  {
745  PromptMenu();
746  }
747  }
748  // Completion so far.
749  else if (B_Completion && (! str_CompletionSoFar.IsEmpty()))
750  {
751  m_cliCmdLine.Delete(GetStream(ECHO_STREAM), - cli_CommandLine.GetNumBackspacesForCompletion());
752  m_cliCmdLine.Put(GetStream(ECHO_STREAM), str_CompletionSoFar);
753  }
754  }
755  else
756  {
757  if (B_PromptMenu)
758  {
759  m_cliCmdLine.NextLine(GetStream(ECHO_STREAM));
760  }
761  PrintError(GetInput().GetLocation(), cli_CommandLine.GetLastError());
762  if (B_PromptMenu)
763  {
764  PromptMenu();
765  }
766  }
767  }

770 void Shell::OnExecute(void)
771 {
772  m_cliCmdLine.NextLine(GetStream(ECHO_STREAM));
773 
774  if (! m_qMenus.IsEmpty())
775  {
776  // First of all, store the line in history stack.
777  m_cliHistory.Push(m_cliCmdLine);
778 
779  // Then check whether this is a comment line.
780  bool b_CommentLine = false;
781  for ( tk::Queue<tk::String>::Iterator it0 = GetCli().GetCommentLinePatterns().GetIterator();
782  (! b_CommentLine) && GetCli().GetCommentLinePatterns().IsValid(it0);
783  GetCli().GetCommentLinePatterns().MoveNext(it0))
784  {
785  const tk::String& tk_Pattern = GetCli().GetCommentLinePatterns().GetAt(it0);
786  if (m_cliCmdLine.GetLine().SubString(0, tk_Pattern.GetLength()) == tk_Pattern)
787  {
788  b_CommentLine = true;
789  }
790  }
791 
792  // Non-comment lines.
793  if (! b_CommentLine)
794  {
795  // Append "\n" to the line and parse it.
796  tk::String str_Line(MAX_CMD_LINE_LENGTH + 1);
797  CommandLine cli_CommandLine;
798  if (str_Line.Set(m_cliCmdLine.GetLine()) && str_Line.Append("\n")
799  && cli_CommandLine.Parse(*m_qMenus.GetTail(), str_Line, true))
800  {
801  bool b_Executed = false;
802  // No need to execute unless there are at least one word and one endl elements in the command line.
803  CommandLineIterator it(cli_CommandLine);
804  if (it.StepIt() && it.StepIt())
805  {
806  // Command line execution.
807  if (const Menu* const pcli_Menu = m_qMenus.GetTail())
808  {
809  if (pcli_Menu->ExecuteReserved(cli_CommandLine))
810  {
811  b_Executed = true;
812  }
813  if (pcli_Menu->Execute(cli_CommandLine))
814  {
815  b_Executed = true;
816  }
817  }
818 
819  // Post-execution.
820  if (b_Executed)
821  {
822  // Entering a sub-menu.
823  if (const Endl* pcli_Endl = dynamic_cast<const Endl*>(& cli_CommandLine.GetLastElement()))
824  {
825  if (pcli_Endl->GetMenuRef() != NULL)
826  {
827  EnterMenu(pcli_Endl->GetMenuRef()->GetMenu(), false);
828  }
829  }
830  }
831  else
832  {
833  const ResourceString cli_ExecutionError = ResourceString()
834  .SetString(ResourceString::LANG_EN, "Execution error")
835  .SetString(ResourceString::LANG_FR, "Erreur d'excution");
836  PrintError(GetInput().GetLocation(), cli_ExecutionError);
837  }
838  }
839  }
840  else
841  {
842  PrintError(GetInput().GetLocation(), cli_CommandLine.GetLastError());
843  }
844  }
845 
846  // Let's get ready for another command line.
847  m_cliCmdLine.Reset();
848  PromptMenu();
849  }

861 void Shell::PrintHelp(const Element& CLI_Element)
862 {

864  tk::String str_Keyword = CLI_Element.GetKeyword();
865  tk::String str_Help = CLI_Element.GetHelp().GetString(GetLang());
866 
867  // Optional corrections.

869  {
870  str_Keyword.Set("<cr>");
871  }

875  {

879  if (! pcli_Param->GetstrValue().IsEmpty())
880  {
881  if (! str_Keyword.Append("("))
882  GetTraces().Trace(INTERNAL_ERROR) << "Shell::PrintHelp(): Could not review str_Keyword for the parameter" << endl;
883  if (! str_Keyword.Append(pcli_Param->GetstrValue()))
884  GetTraces().Trace(INTERNAL_ERROR) << "Shell::PrintHelp(): Could not review str_Keyword for the parameter" << endl;
885  if (! str_Keyword.Append(")"))
886  GetTraces().Trace(INTERNAL_ERROR) << "Shell::PrintHelp(): Could not review str_Keyword for the parameter" << endl;
887  }
888  delete pcli_Clone;
889  }
890  }
891 
892  // Display.
893  unsigned int i = 0;
894  for (i = 0; i < HELP_MARGIN; i++)
895  {
896  GetStream(OUTPUT_STREAM) << " ";
897  }
898  GetStream(OUTPUT_STREAM) << str_Keyword << " ";
899  for (i = str_Keyword.GetLength(); i < HELP_OFFSET; i++)
900  {
901  GetStream(OUTPUT_STREAM) << " ";
902  }
903  GetStream(OUTPUT_STREAM) << str_Help;
904  GetStream(OUTPUT_STREAM) << endl;
