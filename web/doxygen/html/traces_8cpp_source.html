150 const bool Traces::SetStream(OutputDevice& CLI_Stream)
151 {
152  // Store next reference.
153  CLI_Stream.UseInstance(__CALL_INFO__);
154  if (! CLI_Stream.OpenUp(__CALL_INFO__))
155  {
156  OutputDevice::GetStdErr() << CLI_Stream.GetLastError().GetString(ResourceString::LANG_DEFAULT) << endl;
157 
158  // Store nothing on error.
159  CLI_Stream.FreeInstance(__CALL_INFO__);
160  return false;
161  }
162  // Do not store the reference until opening is done.
163  m_qStreams.AddHead(& CLI_Stream);
164 
165  return true;
166 }

168 const bool Traces::UnsetStream(OutputDevice& CLI_Stream)
169 {
170  for ( tk::Queue<OutputDevice*>::Iterator it = m_qStreams.GetIterator();
171  m_qStreams.IsValid(it);
172  m_qStreams.MoveNext(it))
173  {
174  if (OutputDevice* const pcli_Stream = m_qStreams.GetAt(it))
175  {
176  if (pcli_Stream == & CLI_Stream)
177  {
178  bool b_Res = true;
179 
180  // Unreference the device right now.
181  m_qStreams.Remove(it);
182  if (! pcli_Stream->CloseDown(__CALL_INFO__))
183  {
184  OutputDevice::GetStdErr() << pcli_Stream->GetLastError().GetString(ResourceString::LANG_DEFAULT) << endl;
185  b_Res = false;
186  }
187  pcli_Stream->FreeInstance(__CALL_INFO__);
188 
189  if (! b_Res)
190  {
191  // Abort on error
192  return false;
193  }
194  return true;
195  }
196  }
197  }
198 
199  return false;
200 }

202 const bool Traces::Declare(const TraceClass& CLI_Class)
203 {
204  if (m_mapClasses.GetAt(CLI_Class.GetName()) != NULL)
205  {
206  // Already declared.
207  // Nothing else to do.
208  return true;
209  }
210  else
211  {
212  // Show this new trace class depending on the trace all configuration.
213  const bool b_ShowTrace = m_bTraceAll;
214 
215  // Remember this new class.
216  if (m_mapClasses.SetAt(CLI_Class.GetName(), TraceClassFlag(CLI_Class, b_ShowTrace)))
217  {
218  TraceFilterState(CLI_Class, b_ShowTrace);
219  return true;
220  }
221  else
222  {
223  return false;
224  }

271 const bool Traces::SetFilter(const TraceClass& CLI_Class, const bool B_ShowTraces)
272 {
273  // Check the class exists.
274  if (const TraceClassFlag* const pcli_Flags = m_mapClasses.GetAt(CLI_Class.GetName()))
275  {
276  if (pcli_Flags->IsVisible() != B_ShowTraces)
277  {
278  // Set the flag.
279  m_mapClasses.SetAt(CLI_Class.GetName(), TraceClassFlag(*pcli_Flags, B_ShowTraces));
280  TraceFilterState(CLI_Class, B_ShowTraces);
281  }
282  return true;
283  }
284 
285  return false;

288 const bool Traces::SetAllFilter(const bool B_ShowTraces)
289 {
290  // Remember 'all filter configuration'.
291  m_bTraceAll = B_ShowTraces;
292  BeginTrace(TRACE_TRACES) << "All traces " << (B_ShowTraces ? "enabled" : "disabled") << endl;
293 
294  // Trace traces modifications.
295  if (! SetFilter(TRACE_TRACES, true))
296  {
297  return false;
298  }
299 
300  // Then adapt every filter configuration.
301  for ( ClassMap::Iterator it = m_mapClasses.GetIterator();
302  m_mapClasses.IsValid(it);
303  m_mapClasses.MoveNext(it))
304  {
305  if (! SetFilter(m_mapClasses.GetAt(it), B_ShowTraces))
306  {
307  return false;
308  }
309  }
310 
311  // Set final traces settings.
312  SetFilter(TRACE_TRACES, B_ShowTraces);
313 
314  return true;
315 }
